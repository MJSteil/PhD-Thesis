}

\newcommand{\cventry}[4]{
	\par\noindent%
	\begin{tabular}{@{}m{0.24\textwidth}@{\hspace{0.035\textwidth}}p{0.725\textwidth}@{}}

		& #4  \\
	\end{tabular}\vspace{0.2em}
	\par
}
