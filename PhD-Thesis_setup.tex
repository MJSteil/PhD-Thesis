%% Typesetting, see, e.g., [https://ctan.org/pkg/microtype] and [https://www.khirevich.com/latex/microtype/]
% The following typesetting setup implements better linespacing and breaking especially around \dash{} and when trying to prevent line breaks in specific expressions
\usepackage{microtype} % "Subliminal refinements towards typographical perfection"
\microtypesetup{activate={true,nocompatibility},tracking=true,kerning=true,spacing=true,factor=750,stretch=20,shrink=20}
% activate={true,nocompatibility} - activate protrusion and expansion
% tracking=true, kerning=true, spacing=true - activate these techniques
% factor=750 -- protrusion amount (default is 1000):  750 reduces/pentaltizes protrusion a bit more
% stretch=..., shrink=... - stretchability/shrinkability (default is 20/20)
\microtypecontext{spacing=nonfrench} % prevents a corresponding warning
\SetTracking{encoding={*}, shape=sc}{30} % reduce tracking for small caps to 0.03*1em [https://www.khirevich.com/latex/microtype/]
\WarningFilter{microtype}{I cannot find a spacing list for font} % Disable warning for some missing XCharter sizes/fonts
